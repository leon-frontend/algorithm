### _**电话号码的字母组合**_

> _**问题描述**_
>
> - 给定一个仅包含数字 `2-9` 的字符串，返回所有它能表示的字母组合。答案可以按 **任意顺序** 返回。给出数字到字母的映射如下（与电话按键相同）。注意 **1** 不对应任何字母。
> 
> ![[回溯 - 9.png|Untitled 84.png]]
>
> ```JavaScript
> var letterCombinations = function(digits) {
>   // 结果返回数组
>   const res = []
>   // 如果字符串为空，直接返回空数组
>   if (digits.length === 0) return res
>   const phoneMap = { 
> 	  '2': 'abc','3': 'def','4': 'ghi','5': 'jkl','6': 'mno','7': 'pqrs','8': 'tuv','9': 'wxyz'
>   }
> 
>   // 回溯
>   // index 参数表示当前正在处理的数字串 digits 的索引
>   // state 表示当前状态，是一个字符
>   const backtrack = (index, state) => {
>     // 当 当前处理的字符 是 digits 的最后一个字符时，记录解
>     if (index === digits.length) {
>       res.push(state)
>       return
>     }
>     // 尝试：作出选择，选择数字
>     const num = digits.charAt(index)
>     // 选择数字对应的字符串
>     const chars = phoneMap[num]
>     // 遍历所有字符串中的所有字符
>     for (let i = 0; i < chars.length; i++) {
>       // 尝试：做出选择，更新状态
>       state += chars.charAt(i)
>       // 递归
>       backtrack(index + 1, state)
>       // 回退：删除最后一个字符
>       state = state.substring(0, state.length - 1)
>     }
>   }
>   backtrack(0, '')
>   return res
> };
> ```

>     if (state.length === max * 2) {
>       res.push(state)
>       return
>     }
> 
>     // 剪枝 1：左括号的数量一定要小于 n 的大小
>     if (left < max) {
>       // // 尝试：作出选择，更新状态
>       // state += '('
>       // left += 1
>       // // 递归
>       // backtrack(state, max, left, right)
>       // // 回退
>       // state = state.substring(0, state.length - 1)

>     if (left > right) {

### _**单词搜索**_

> _**问题描述**_
>
> > [!important] **[LeetCode - 单词搜索](https://leetcode.cn/problems/word-search/description/?envType=study-plan-v2&envId=top-100-liked)**
>
> - 给定一个 `m x n` 二维字符网格 `board` 和一个字符串单词 `word`。如果 `word` 存在于网格中，返回 `true` ；否则，返回 `false` 。单词必须按照字母顺序，通过相邻的单元格内的字母构成，其中“相邻”单元格是那些水平相邻或垂直相邻的单元格。同一个单元格内的字母不允许被重复使用。
> 
> ![[回溯 - 11.png|Untitled 2 51.png]]
>
> ```JavaScript
> var exist = function(board, word) {
>   const rows = board.length  // 获取表格行数
>   const cols = board[0].length  // 获取表格列数
> 
>   // 定义 dfs + 回溯 算法
>   // index 表示 word 的下标索引
>   const backtrack = (row, col, index) => {
>     // 1. 记录解的三种情况，剪枝
>     // 1.1 成功找到单词
>     if (word.length === index) return true
>     // 1.2 越界，返回 false
>     if (row < 0 || row >= rows || col < 0 || col >= cols) return false
>     // 1.3 与 word 不匹配
>     if (board[row][col] !== word.charAt(index)) return false
> 
>     // 2. 尝试：标记当前匹配的字符，以避免在同一路径中重复使用
>     board[row][col] = '#'
> 
>     // 3. 递归进行下一次选择
>     const res = backtrack(row + 1, col, index + 1) || // 下
>                 backtrack(row - 1, col, index + 1) || // 上
>                 backtrack(row, col + 1, index + 1) || // 右
>                 backtrack(row, col - 1, index + 1)    // 左
> 
>     // 4. 回溯，还原单元格
>     board[row][col] = word.charAt(index)
>     return res
>   }
> 
>   // 遍历 board 中的每一个元素
>   for (let row = 0; row < rows; row++) {
>     for (let col = 0; col < cols; col++) {
>       if (backtrack(row, col, 0)) return true
>     }
>   }
>   return false
> };
> ```

### _**分割回文串**_

> _**问题描述**_
>
> > [!important] **[LeetCode - 分割回文串](https://leetcode.cn/problems/palindrome-partitioning/description/?envType=study-plan-v2&envId=top-100-liked)**
>
> - 给你一个字符串 `s`，请你将 `s` 分割成一些子串，使每个子串都是 **回文串** 。返回 `s` 所有可能的分割方案。**回文串** 是正着读和反着读都一样的字符串

> _**算法分析 ⇒ 本质是子集问题**_
>
> - 从字符串的开始位置遍历到结束位置，逐步增加当前考虑的子串长度
> 
> - 对于每个子串，检查它是否是回文串
>     
>     - 如果是，将其加入到当前分割方案中 并递归地继续分割剩余字符串
>     
>     - 如果不是，跳过当前子串，继续尝试下一个可能的分割
>     
> 
> - 当到达字符串的末尾时，将当前分割方案加入到结果列表中
> 
> - 使用回溯法在每次递归返回后撤销最后一次添加的子串，以尝试新的分割方案
> 
> ![[回溯 - 13.png|Untitled 4 30.png]]
>
> ```JavaScript
> 
> var partition = function(s) {
>   const state = [] // 声明状态数组
>   const res = [] // 声明结果数组
> 
>   // 判断一个字符串是否回文
>   const isPalindrome = (start, end) => {
>     while (start < end) {
>       if (s[start] !== s[end]) return false
>       start++
>       end--
>     }
>     return true
>   }
> 
>   // 回溯算法
>   const backtrack = (start) => {
>     // 记录解
>     if (start >= s.length) {
>       res.push([...state])
>       return
>     }
> 
>     // 分割字符串中的每一个字符
>     for (let end = start; end < s.length; end++) {
>       if (isPalindrome(start, end)) {
>         // 作出选择
>         state.push(s.substring(start, end + 1))
>         // 递归
>         backtrack(end + 1)
>         // 回退
>         state.pop()
>       }
>     }
>   }
> 
>   backtrack(0)
>   return res
> };
> ```
